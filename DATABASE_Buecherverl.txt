);


CREATE TABLE kunde (
    nummer CHAR(10) PRIMARY KEY,
    name VARCHAR(200) NOT NULL,
    vorname VARCHAR(200) NOT NULL,
    guthaben DOUBLE NOT NULL
);


CREATE TABLE ausleihe (
    ausleihe_id INT PRIMARY KEY AUTO_INCREMENT,
    buch_id INT,
    kundennummer CHAR(10),
